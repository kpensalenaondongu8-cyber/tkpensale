package piscine

import "strconv"

func ZipString(s string) string {
    if len(s) == 0 {
        return ""
    }
    result := ""
    count := 1
    for i := 0; i < len(s); i++ {
        if i+1 < len(s) && s[i] == s[i+1] {
            count++
        } else {
            result += strconv.Itoa(count) + string(s[i])
            count = 1
        }
    }
    return result
}
