package main

import (
    "os"
    "strings"
)

func main() {
    if len(os.Args) != 2 {
        return
    }
    s := os.Args[1]
    if s == "" {
        return
    }
    words := strings.Split(s, " ")
    for i := len(words) - 1; i >= 0; i-- {
        os.Stdout.WriteString(words[i])
        if i > 0 {
            os.Stdout.WriteString(" ")
        }
    }
    os.Stdout.WriteString("\n")
}
