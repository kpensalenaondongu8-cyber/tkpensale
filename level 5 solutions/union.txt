package main

import (
    "os"
)

func main() {
    if len(os.Args) != 3 {
        os.Stdout.WriteString("\n")
        return
    }
    s1, s2 := os.Args[1], os.Args[2]
    seen := make(map[byte]bool)
    for _, c := range s1 + s2 {
        if !seen[byte(c)] {
            os.Stdout.WriteString(string(c))
            seen[byte(c)] = true
        }
    }
    os.Stdout.WriteString("\n")
}
