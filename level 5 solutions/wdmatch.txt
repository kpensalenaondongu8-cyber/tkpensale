package main

import (
    "os"
)

func main() {
    if len(os.Args) != 3 {
        return
    }
    s1, s2 := os.Args[1], os.Args[2]
    i := 0
    for _, c := range s2 {
        if i < len(s1) && byte(c) == s1[i] {
            i++
        }
    }
    if i == len(s1) {
        os.Stdout.WriteString(s1 + "\n")
    }
}
